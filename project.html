FINAL PROJECT(3D GRAPHIC TRANSFORMATIONS)

I. PROJECT OVERVIEW

This project showcases how 3D geometric transformations can be applied to a music disc model using Python and VPython. By animating the disc in a virtual environment, it becomes easier to visualize how objects move and change within three-dimensional space.

Throughout the demonstration, the disc is translated, rotated, and scaled to highlight the core principles of linear algebra. These transformations help illustrate how mathematical operations can manipulate objects in 3D, making the concepts more intuitive and engaging.

II. METHODOLOGY

1. Created a 3D Black Music Disc using Vpython shapes.

2. Set up a helper functions for vector and matrix operation.

3. Applied translation using a smooth movement.

4. Implemented rotation around the x-axis and y-axis using NumPy rotation matrices.

5. Applied scaling by adjusting the radius of all disc components to resize the object.

6. Demonstrated transformations in sequence.

# === Helper functions ===

def to_np(v):

"""Convert VPython vector → NumPy array."""

if isinstance(v, vector):

return np.array([v.x, v.y, v.z])

elif isinstance(v, (list, tuple, np.ndarray)):

return np.array(v)

else:

raise TypeError("Unsupported type for to_np()")

def to_vpy(v):

"""Convert NumPy array → VPython vector."""

return vector(float(v[0]), float(v[1]), float(v[2]))

def get_ry(theta):

"""Rotation matrix around Y-axis."""

return np.array([

[np.cos(theta), 0, np.sin(theta)],

[0, 1, 0],

[-np.sin(theta), 0, np.cos(theta)]

])

def get_rx(theta):

"""Rotation matrix around X-axis"""

return np.array([

[1, 0, 0],

[0, np.cos(theta), -np.sin(theta)],

[0, np.sin(theta), np.cos(theta)]

])

# === TRANSFORMATIONS ===

# Translation

def smooth_move(target_pos, steps=100, wait=1):

"""Translate all shapes smoothly together."""

target_np = to_np(target_pos)

start_positions = [to_np(s.pos) for s in myshapes_1]

reference_start = start_positions[0]

move_vector = target_np - reference_start

for t in np.linspace(0, 1, steps):

rate(60)

for s, start in zip(myshapes_1, start_positions):

new_v = start + move_vector * t

s.pos = to_vpy(new_v)

# Scaling

def smooth_scale_all(scale_factor, steps=60):

for t in np.linspace(0, 1, steps):

rate(60)

for s in myshapes_1:

factor = 1 + (scale_factor -1)/steps

if hasattr(s, 'radius'):

s.radius *= factor

# Rotation

def smooth_rotate(theta, steps=100, wait=1):

"""Rotate all shapes smoothly around X-axis."""

Rx = get_rx(np.radians(theta / steps))

for _ in range(steps):

rate(60)

for s in myshapes_1:

v = to_np(s.pos)

v = Rx @ v

s.pos = to_vpy(v)

s.axis = to_vpy(Rx @ to_np(s.axis))

def smooth_rotate_y(theta, steps=100, wait=1):

"""Rotate all shapes smoothly around Y-axis."""

Ry = get_ry(np.radians(theta / steps))

for _ in range(steps):

rate(60)

for s in myshapes_1:

v = to_np(s.pos)

v = Ry @ v

s.pos = to_vpy(v)

s.axis = to_vpy(Ry @ to_np(s.axis))
